Dependency Information

Apache Maven

<dependency>
  <groupId>fr.fastconnect.factory.tibco.bw.codereview</groupId>
  <artifactId>code-review-plugin</artifactId>
  <version>2.0.0-SNAPSHOT</version>
  <type>maven-plugin</type>
</dependency>

Apache Buildr

'fr.fastconnect.factory.tibco.bw.codereview:code-review-plugin:maven-plugin:2.0.0-SNAPSHOT'

Apache Ant

<dependency org="fr.fastconnect.factory.tibco.bw.codereview" name="code-review-plugin" rev="2.0.0-SNAPSHOT">
  <artifact name="code-review-plugin" type="maven-plugin" />
</dependency>

Groovy Grape

@Grapes(
@Grab(group='fr.fastconnect.factory.tibco.bw.codereview', module='code-review-plugin', version='2.0.0-SNAPSHOT')
)

Grails

compile 'fr.fastconnect.factory.tibco.bw.codereview:code-review-plugin:2.0.0-SNAPSHOT'

Leiningen

[fr.fastconnect.factory.tibco.bw.codereview/code-review-plugin "2.0.0-SNAPSHOT"]

SBT

libraryDependencies += "fr.fastconnect.factory.tibco.bw.codereview" %% "code-review-plugin" % "2.0.0-SNAPSHOT"
