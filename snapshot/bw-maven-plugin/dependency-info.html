Maven plugin for TIBCO BusinessWorks

Enable Java-like lifecycle management in TIBCO BusinessWorks

Dependency Information

Apache Maven

<dependency>
  <groupId>fr.fastconnect.factory.tibco.bw.maven</groupId>
  <artifactId>bw-maven-plugin</artifactId>
  <version>2.4.2-SNAPSHOT</version>
  <type>maven-plugin</type>
</dependency>

Apache Buildr

'fr.fastconnect.factory.tibco.bw.maven:bw-maven-plugin:maven-plugin:2.4.2-SNAPSHOT'

Apache Ant

<dependency org="fr.fastconnect.factory.tibco.bw.maven" name="bw-maven-plugin" rev="2.4.2-SNAPSHOT">
  <artifact name="bw-maven-plugin" type="maven-plugin" />
</dependency>

Groovy Grape

@Grapes(
@Grab(group='fr.fastconnect.factory.tibco.bw.maven', module='bw-maven-plugin', version='2.4.2-SNAPSHOT')
)

Grails

compile 'fr.fastconnect.factory.tibco.bw.maven:bw-maven-plugin:2.4.2-SNAPSHOT'

Leiningen

[fr.fastconnect.factory.tibco.bw.maven/bw-maven-plugin "2.4.2-SNAPSHOT"]

SBT

libraryDependencies += "fr.fastconnect.factory.tibco.bw.maven" %% "bw-maven-plugin" % "2.4.2-SNAPSHOT"
